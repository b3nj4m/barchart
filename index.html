Easy SVG bar-charts.

Easily generate animated SVG barcharts in the browser using d3.js and barchart.js.

requirejs.config({
  shim: {
    'd3': {exports: 'd3'}
  }
});
define(['barchart'], function(BarChart) {
  var bc = new BarChart({
    data: [[
      {"name": "#1", "value": 12},
      {"name": "#2", "value": 20},
      {"name": "#3", "value": 30},
      {"name": "#4", "value": 70},
      {"name": "#5", "value": 63},
      {"name": "#6", "value": 35}
    ],[
      {"name": "#1'", "value": 2},
      {"name": "#2'", "value": 10},
      {"name": "#3'", "value": 3},
      {"name": "#4'", "value": 7},
      {"name": "#5'", "value": 6},
      {"name": "#6'", "value": 3}
    ]],
    bar_colors: ['#00AB8E', '#33CCDD'],
    label_inside_colors: ['#FFF', '#333'],
    auto_scale: true,
    container: document.getElementById('chart-container')
  });
  bc.render();
});

Config Options

var bc = new BarChart({
  //DOM element to render the chart into
  container: null,

  //set to false if you don't want animation
  is_animated: true,

  //animation duration in ms
  animation_duration: 400,

  //whether or not BarChart should scale to the width of the container
  auto_scale: false,

  //which of d3's scale types to use
  height_scale_type: 'linear',

  //color to fill the bars with (also accepts an array for multi-dataset support)
  bar_colors: '#00AB8E',

  //distance between each bar
  bar_spacing: 20,

  //padding around the outer edges of the chart
  chart_padding: 10,

  //the data from which the chart will be rendered
  data: [],

  //the key to use when looking up a value on an item in the data array
  data_value_key: 'value',

  //default height of the chart in pixels
  height: 300,

  //default width of the chart if not using auto_scale
  width: 700,

  //text color of the top labels (also accepts an array for multi-dataset support)
  label_top_colors: '#003D4C',

  //text color of the labels inside the bars (also accepts an array for multi-dataset support)
  label_inside_colors: '#FFF',

  //should usually be the same as the data_value_key
  label_inside_key: 'value',

  //padding in pixels around each label
  label_padding: 3,

  //font size of the label texts
  label_size: 16,

  //the key to use when looking up the top label (usually a name) on an item in the data array
  label_top_key: 'name'
});
